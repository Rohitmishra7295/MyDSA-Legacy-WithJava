class Solution {
    public int[] plusOne(int[] digits) {
        int n = digits.length;

        for (int i = n - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            digits[i] = 0;
        }

        // If all digits were 9 (e.g., 999 + 1 = 1000)
        int[] result = new int[n + 1];
        result[0] = 1; // e.g., 1000 â†’ [1, 0, 0, 0]
        return result;
    }
}
